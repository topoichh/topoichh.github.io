princas = lygis12.princas
princas.rodykVektorių()

komanda Forward(mygtukas):
    princas.pirmyn()

komanda Backward(mygtukas):
    princas.atgal()

komanda Left(mygtukas):
    princas.kairėn()

komanda Right(mygtukas):
    princas.dešinėn()

komanda Stop(mygtukas):
    princas.stok()

komanda Attack():
    princas.kirsk()


žaidimas.mygtukąPaspaudus('w', Forward)
žaidimas.mygtukąPaspaudus('s', Backward)
žaidimas.mygtukąPaspaudus('a', Left)
žaidimas.mygtukąPaspaudus('d', Right)

žaidimas.mygtukąAtleidus('<betKuris>', Stop)


žaidimas.pelęPaspaudus(Attack)
